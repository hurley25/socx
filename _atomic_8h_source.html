27 /**
28  * 原子整数
29  */
30 template <typename T>
31 class AtomicInt : private noncopyable
32 {
33 public:
34  AtomicInt() : value_(0) { }
35 
36  /// 获得原子变量的值
37  T get()
38  {
39  // in g++ >= 4.7: __atomic_load_n(&value_, __ATOMIC_SEQ_CST)
40  return __sync_val_compare_and_swap(&value_, 0, 0);
41  }
42 
43  /// 原子变量Add操作并后返原值
44  T getAndAdd(T x)
45  {
46  return __sync_fetch_and_add(&value_, x);
47  }
48 
49  /// 原子变量Add操作并返回Add后的值
50  T addAndGet(T x)
51  {
52  return getAndAdd(x) + x;
53  }
54 
55  /// 原子变量自增1并返回原值
56  T incrementAndGet()
57  {
58  return addAndGet(1);
59  }
60 
61  /// 原子变量自减1并返回原值
62  T decrementAndGet()
63  {
64  return addAndGet(-1);
65  }
66 
67  /// 原子变量Add
68  void add(T x)
69  {
70  getAndAdd(x);
71  }
72 
73  /// 原子变量自增1
74  void increment()
75  {
76  incrementAndGet();
77  }
78 
79  /// 原子变量自减1
80  void decrement()
81  {
82  decrementAndGet();
83  }
84 
85  /// 原子变量设置为新值并返回旧值
86  T getAndSet(T newValue)
87  {
88  return __sync_lock_test_and_set(&value_, newValue);
89  }
90 
91 private:
92  volatile T value_;
93 };
