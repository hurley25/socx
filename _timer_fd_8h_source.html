28 /**
29  * timerfd 定时器的封装
30  */
31 class TimerFd : private noncopyable
32 {
33 public:
34  /// 构造函数默认创建非阻塞的 timerfd
35  TimerFd(int fd_flags = TFD_NONBLOCK)
36  {
37  timer_fd_ = timerfd_create(CLOCK_REALTIME, fd_flags);
38  CHECK(timer_fd_ != 0) << "create timer_fd error, errno: " << errno << " msg:" << strerror(errno);
39  }
40 
41  ~TimerFd()
42  {
43  close(timer_fd_);
44  }
45 
46  /// 启动循环触发的毫秒级别定时器
47  int startIntervalMSecTimer(long msec)
48  {
49  struct timespec tv;
50  int ret = clock_gettime(CLOCK_REALTIME, &tv);
51  if (ret == -1) {
52  return ret;
53  }
54 
55  struct itimerspec itv;
56 
57  long sec = msec / 1000;
58  long nsec = (msec % 1000) * 1000 * 1000;
59 
60  itv.it_value.tv_sec = tv.tv_sec + sec;
61  itv.it_value.tv_nsec = tv.tv_nsec + nsec;
62  itv.it_interval.tv_sec = sec;
63  itv.it_interval.tv_nsec = nsec;
64 
65  return timerfd_settime(timer_fd_, TFD_TIMER_ABSTIME, &itv, NULL);
66  }
67 
68  /// 启动循环触发的秒级别定时器
69  int startIntervalSecTimer(long sec)
70  {
71  return startIntervalMSecTimer(sec * 1000);
72  }
73 
74  /// 启动一次性的毫秒级别定时器
75  int startOnceMSecTimer(long msec)
76  {
77  struct timespec tv;
78  int ret = clock_gettime(CLOCK_REALTIME, &tv);
79  if (ret == -1) {
80  return ret;
81  }
82 
83  struct itimerspec itv;
84  bzero(&itv, sizeof(itv));
85 
86  long sec = msec / 1000;
87  long nsec = (msec % 1000) * 1000 * 1000;
88 
89  itv.it_value.tv_sec = tv.tv_sec + sec;
90  itv.it_value.tv_nsec = tv.tv_nsec + nsec;
91 
92  return timerfd_settime(timer_fd_, TFD_TIMER_ABSTIME, &itv, NULL);
93  }
94 
95  /// 启动一次性的秒级别定时器
96  int startOnceSecTimer(long sec)
97  {
98  return startOnceMSecTimer(sec * 1000);
99  }
100 
101  /// 启动一次性的绝对时间触发定时器
102  int startAbsTimer(struct timespec &tv)
103  {
104  struct itimerspec itv;
105  bzero(&itv, sizeof(itv));
106  itv.it_value = tv;
107 
108  return timerfd_settime(timer_fd_, TFD_TIMER_ABSTIME, &itv, NULL);
109  }
110 
111  /// 停止定时器时间（针对循环定时器）
112  int stopTimer()
113  {
114  struct itimerspec itv;
115  bzero(&itv, sizeof(itv));
116  return timerfd_settime(timer_fd_, TFD_TIMER_ABSTIME, &itv, NULL);
117  }
118 
119  /// 获得内部的 timerfd 文件描述符
120  int fd() const
121  {
122  return timer_fd_;
123  }
124 
125 private:
126  int timer_fd_;
127 };
