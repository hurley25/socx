35  uint64_t rdtsc;
36  __asm__ __volatile__("rdtsc" : "=A" (rdtsc));

40  /// 取得当前时间的 ns 值
41  static inline uint64_t nowNs()
42  {
43  struct timespec tv;
44  int ret = clock_gettime(CLOCK_REALTIME, &tv);
45  CHECK(ret != -1) << "clock_gettime error, errno: " << errno << " msg:" << strerror(errno);
46  return (tv.tv_sec * (uint64_t)1000000000 + tv.tv_nsec);
47  }
48 
49  /// 取得当前时间的 us 值
50  static inline uint64_t nowUs()
51  {
52  struct timespec tv;
53  int ret = clock_gettime(CLOCK_REALTIME, &tv);
54  CHECK(ret != -1) << "clock_gettime error, errno: " << errno << " msg:" << strerror(errno);
55  return (tv.tv_sec * (uint64_t)1000000 + tv.tv_nsec / 1000);
56  }
57 
58  /// 取得当前时间的 ms 值
59  static inline uint64_t nowMs()
60  {
61  struct timespec tv;
62  int ret = clock_gettime(CLOCK_REALTIME, &tv);
63  CHECK(ret != -1) << "clock_gettime error, errno: " << errno << " msg:" << strerror(errno);
64  return (tv.tv_sec * (uint64_t)1000 + tv.tv_nsec / 1000000);
65  }
66 };
