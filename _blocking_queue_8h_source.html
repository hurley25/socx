29 /**
30  * 无界阻塞队列(生产者/消费者队列)
31  */
32 template <typename T>
33 class BlockingQueue : private noncopyable
34 {
35 public:
36  BlockingQueue() { }
37  ~BlockingQueue() { }
38 
39  /// 添加任务
40  void put(T task)
41  {
42  std::lock_guard<std::mutex> lock(mutex_);
43  queue_.push_back(task);
44  notEmptyCond_.notify_one();
45  }
46 
47  /// 取得任务
48  T take()
49  {
50  std::unique_lock<std::mutex> lock(mutex_);
51  // 避免虚假唤醒
52  while (queue_.empty()) {
53  notEmptyCond_.wait(lock);
54  }
55  T task(queue_.front());
56  queue_.pop_front();
57 
58  return task;
59  }
60 
61  /// 获得队列长度
62  size_t size() const
63  {
64  std::lock_guard<std::mutex> lock(mutex_);
65 
66  return queue_.size();
67  }
68 
69  /// 判断队列是否为空
70  bool empty() const
71  {
72  std::lock_guard<std::mutex> lock(mutex_);
73 
74  return queue_.empty();
75  }
76 
77 private:
78  mutable std::mutex mutex_;
79  std::condition_variable notEmptyCond_;
80  std::deque<T> queue_;
81 };
