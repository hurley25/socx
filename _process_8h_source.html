60  /// 获得当前进程启动时间
61  static inline int getStartTime(std::string &stime)
62  {
63  char time_buff[20];
64  struct stat st;
65  int ret = ::lstat("/proc/self/exe", &st);
66  if (ret < 0) {
67  return ret;
68  }
69 
70  struct tm *tm_loc = localtime(&st.st_mtime);
71  snprintf(time_buff, sizeof(time_buff), "%04d-%02d-%02d %02d:%02d:%02d",
72  tm_loc->tm_year + 1900, tm_loc->tm_mon + 1, tm_loc->tm_mday,
73  tm_loc->tm_hour, tm_loc->tm_min, tm_loc->tm_sec);
74 
75  stime = time_buff;
76 
77  return 0;
78  }
79 
80  /// 获得当前进程的可执行文件路径
81  static inline int getExeFilePath(std::string &path)
82  {
83  char path_buf[PATH_MAX];
84  ssize_t ret = readlink("/proc/self/exe", path_buf, sizeof(path_buf));
85  if (ret > 0) {
86  path.assign(path_buf, ret);
87  return 0;
88  }
89 
90  return static_cast<int>(ret);
